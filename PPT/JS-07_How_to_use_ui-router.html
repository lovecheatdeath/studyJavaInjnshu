1.背景介绍

单页面应用、路由？

当我们的应用越来越多时，切换起来没那么流畅。又是碍于网速，页面加载进来有点慢。这时候以angular等框架开发的单页面应用的优势就体现出来了。 特点是加载次数少，加载以后性能较高，不利于seo（搜索引擎优化）。

路由就是用来定义切换的规则，不用再发起HTTP请求。

UI-Router注入与配置

                
var app=angular.module('myApp',["ui.router"])
    .config(function($stateProvider,$urlRouterProvider){//两个子对象作为参数添加到config对象中去
        $urlRouterProvider.when("","/mainll");//默认路径配置

        /*$urlRouteProvider.when(whenPath, toPath)
         $urlRouterProvider.otherwise(path)
         $urlRouteProvider.rule(handler)*/

        /*$stateProvider.state(stateName, stateConfig);* /  //前一个参数为字符串，后一个为对象

        $stateProvider//应用状态配置
            .state("main", {//状态名
                url:"/mainll",//路径
                templateUrl:"main.html"//HTML模板的路径或者返回HTML模板路径的函数
                controller()//控制器
            })
            .state("main_page",{
                url:"/main_page",
                templateUrl:"main_page.html",
                controller:("myCtrl")
            });
    })
