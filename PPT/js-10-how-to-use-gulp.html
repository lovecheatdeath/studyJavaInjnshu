                    npm install -g gulp
                    //使用npm下载gulp
                    gulp -v

                    进入项目跟目录建立一个package.json
                    //说明：package.json是基于nodejs项目必不可少的配置文件，

                cnpm init -y
                //自动设置配置文件

                gulp.src('*demo/js-10-gulp/demo2.js')
